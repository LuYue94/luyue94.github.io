    1. node 遍历本地文件

    node 遍历本地文件

    一开始用markdown写笔记，并没有找到操作方便，界面又好看的编辑器，于是用atom写。由于我把所有的笔记都分好类别，放在相应的文件夹里面，却没有找到工具，将按照文件目录自动生成对应目录树于是每当新增或者删除markdown文件的时候，都要修改整个项目的catalog.md文件，实在是太过于麻烦，于是写了一个node小插件，在node环境下运行，自动生成对应的markdown文件树列表。 水平有限，智能读取一层目录，多级目录可能需要递归，有空再写。

    var fs = require('fs')


//遍历文件夹，获取所有文件夹里面的文件信息
/*
* @param path 路径
*
*/

function geFileList(path)
{
 var filesList = [];
 var targetObj = {};
 readFile(path,filesList,targetObj);
 return filesList;
}

//遍历读取文件
function readFile(path,filesList,targetObj)
{
 files = fs.readdirSync(path);//需要用到同步读取
 files.forEach(walk);
 function walk(file)
 {  
      states = fs.statSync(path+'/'+file);         
      if(states.isDirectory())
      {
          var item ;
          if(targetObj["children"])
          {
              item = {name:file,children:[]};
              targetObj["children"].push(item);
          }
          else
          {
             item = {name:file,children:[]};
             filesList.push(item);
          }

          readFile(path+'/'+file,filesList,item);
      }
      else
      {   
          //创建一个对象保存信息
          var obj = new Object();
          obj.size = states.size;//文件大小，以字节为单位
          obj.name = file;//文件名
          obj.path = file; //文件绝对路径

          if(targetObj["children"])
          {
             var item = {name:file,value:obj.path}
             targetObj["children"].push(item);
          }
          else
          {
              var item = {name:file,value:obj.path};
              filesList.push(item);
          }
      }     
  }
}

//写入文件utf-8格式
function writeFile(fileName,data)
{  
    fs.writeFile(fileName,data,'utf-8',complete);
    function complete()
    {
       console.log("文件生成成功");
    }
}


var filesList = geFileList("F:/Github/myblog/cherrybooooom.github.io/md");

// 将字符串转成指定的格式
var time = new Date().toLocaleString();
var str = '### Summary\n>LuYue update\n'+ time +'\n\n';
str+= "* [Introduction](README.md)\n";
for (var i = 0; i < filesList.length - 1; i++) {
  if(filesList[i].name.indexOf(".md") == -1){
      var parent = filesList[i].name;
      str+="* ["+ parent +"]("+parent+"/Directory.md)\n"
      for(var j=0;j<filesList[i].children.length;j++){
          var child = filesList[i].children[j].name;
          // 隐藏Directory.md,css,js文件和img文件夹
          var reg = /.js|.css|Directory|img/;
          if(reg.test(child)){
              continue;
          }
          var childDelMd = child.replace(".md","");//去掉md后缀
          str+="    * ["+ childDelMd+"]("+ parent + "/"+ child +")\n";
      }
  }
}

console.log(str);
// 输出文件
writeFile("md/catalog.md",str);
