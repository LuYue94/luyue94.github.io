    工具
    创建时间:2018-03-23 15:13
    阅读:
    1. gulpfile
    2. packgae.json

    在package.json的同级目录下使用命令npm install即可安装所有依赖
    npm start启动node服务
    gulp build启动gulp build任务
    gulp启动gulp watch任务
    所有配置都可按需更改

    gulpfile

    var gulp = require('gulp');
var sass = require("gulp-sass");
// 1.依赖webserver  启动服务
var webserver = require("gulp-webserver");
// 2.mock数据  数据模板 生成模拟数据
var url = require('url');    //node    url信息的读取和设置 路径管路
var fs = require('fs');      // node file层   文件管理模块
//4 js 模块化和打包依赖 gulp-uglify 压缩js文件
var named = require("vinyl-named");
var webpack = require("gulp-webpack");
var uglify  = require('gulp-uglify');
// webserver
gulp.task("webserver",function(){
    gulp.src("./")           // 表示在 根目录启动当前服务器
    .pipe(webserver({
        livereload:true,     //表示浏览器自动刷新  类似热替换
        directoryListing:{   // 表示要不要在浏览器 显示应用的目录结构
            enable:true,     // 表示显示完整的目录结构
            path:"./"        // 表示在根目录显示 当前项目目录结构
        },
        host:"localhost",    // 服务器主机
        port:"8080",                // 端口
           // mock数据  不作了解  就是拦截请求“转发”到本地文件
        middleware: function(req, res, next) {   //middleware 中间键 req :request  , res:respone,
            var urlObj = url.parse(req.url, true),
                method = req.method;
            switch (urlObj.pathname) {
                case '/api/orders':
                    // var data = {
                    //     "status": 0,
                    //     "errmsg": "",
                    //     "data": [{}]
                    // };
                    res.setHeader('Content-Type', 'application/json');
                    fs.readFile('mock/optlist.json','utf-8',function(err,data){
                        res.end(data);
                    });
                    return;
                case '/api/goods':
                    // ...
                    return;
                case '/api/images':
                    // ...
                    return;
                default:
                    ;
            }
            next();
        }
    }))
})

// js入口文件
var jsFiles = ["src/es6/**/*.js"];
/*这里有个小问题，如果只监听entry.js，改变entry.js require的其他js文件，热更新无效，所以要监听所有js文件*/
gulp.task("packJs",function(){
    gulp.src(jsFiles)
    .pipe(named())     // 取出保留js文件的名称
    .pipe(webpack({
        output:{
            filename:'[name].js',    //把取出的js文件名称存入name
        },
        module:{
            loaders:[
                {test:/\.js$/,loader:'imports?define=>false'}   // 如果你js文件不支持commodJS规范，强制转换
            ]
        },
        resolve: {      //别名  文件深，名称较长
        alias: {        //别名

        }
     },
      devtool: '#eval-source-map'   //显示 source-map 子模块的 js调试 debugger
      // 开发的时候启用，方便调试，上线的时候关掉，文件体积太大！
    }))
    .pipe(uglify().on('error', function(e){
      console.log('\x07', e.lineNumber, e.message);
      return this.end();
    }))
    .pipe(gulp.dest("public/js"));
});

var htmlFiles = ["src/**/*.html"];
gulp.task('copy-html',function(){
    return gulp.src(htmlFiles).pipe(gulp.dest('public'))
});

var scssFiles = ["src/scss/**/*.scss"];
gulp.task("sass",function(){
    gulp.src(scssFiles)
    .pipe(sass({outputStyle:"compact"}).on("error",sass.logError))    // 当sass文件编译报错时 不会停止 监听服务，错误部分不编译执行
    .pipe(gulp.dest("public/css"));
});

gulp.task("watch",function(){
    gulp.watch(htmlFiles,['copy-html']);
    gulp.watch(scssFiles,['sass']);
    gulp.watch(jsFiles,['packJs']);
});
gulp.task("build",['moveFiles','move_lib_js','sass','copy-html','packJs']);

gulp.task("default",['build','webserver','watch']);


// 移动lib库的js文件(已经压缩好，不用打包，仅仅是移动位置)
var libFiles = ["src/lib/**/*.js"];
gulp.task('move_lib_js', function() {
    return gulp.src(libFiles)
        // .pipe(concat('main.js'))    //合并所有js到main.js
        .pipe(gulp.dest('public/lib'))    //输出main.js到文件夹
        // .pipe(rename({suffix: '.min'}))   //rename压缩后的文件名
        // .pipe(uglify())    //压缩
        // .pipe(gulp.dest('minified/js'));  //输出
});

//移动img
var imgFiles = ["src/img/**/*"]
gulp.task('moveFiles',function(){
    return gulp.src(imgFiles)
        .pipe(gulp.dest("public/img"))
})
